# Tabs

---

A Tabs component.

## Props

| Name           | Type                 | Default value    | Description                                                                                                                        |
| -------------- | -------------------- | ---------------- | ---------------------------------------------------------------------------------------------------------------------------------- |
| `title`        | **string**           | -                | Title of the tab.                                                                                                                  |
| `customTitle`  | **string**           | -                | If the title needs to be customized. It will be rendered instead of the title.                                                                                                                  |
| `children`     | **ReactNode**        | -                | Content to display on open.                                                                                                        |
| `expanded`     | **boolean*           | false            | Set this as true to show the content by default. If the external prop is the same of the internal state the component dont toggle. |
| `tag`          | **string**           | -                | Text to show as a label right next to the Title.                                                                                   |
| `onToggle`     | **function**         | -                | Function to call when the SlideDown expands or collapses. It returns true if it is expanded and false if it is collapsed.          |
| `textSize`     | **sm | md | lg**     | md               | Size of the Text to show sm, md, lg of the component                                                                               |
| `strong`       | **boolean**          | false            | Set this as true to set title font weight to bold                                                                                  |
| `theme`        | **default | blue**   | default          | Theme of the SlideDown                                                                                                             |
| `noJustified`  | **boolean**          | false            | The tab container flexbox has justify content between by default. Use this property if you need to remove it.                      |
| `divider`      | **boolean**          | false            | Shows a divider at the bottom                                                                                                      |
| `icon`         | **string**           | -                | Use the name of any of the icons available in the library.                                                                         |
| `noPadding`    | **string**           | -                | The container has by default a side padding. Use this property if you need to remove it.                                           |
| `testId`       | **string**           | -                | The testId property adds the data attribute data-testid to the main element and should be used for testing only.                   |

// How to use Tabs Controlled
function TabsMDX() {
  const [value, setValue] = useState<string>('2')

  return (
    <TabsProvider value={value}>
      <Tabs onChange={setValue} size="large">
        <Tab title="Title 1" icon="person" value="1" />
        <Tab title="Title 2" icon="search" value="2" />
        <Tab title="Title 3" icon="search" value="3" />
        <Tab title="Title 4" icon="search" value="4" />
        <Tab title="Title 5" icon="search" value="5" />
        <Tab title="Title 6" icon="search" value="6" />
      </Tabs>
      <TabContent value="1">Contenido 1</TabContent>
      <TabContent value="2">Contenido 2</TabContent>
      <TabContent value="3">Contenido 3</TabContent>
      <TabContent value="4">Contenido 4</TabContent>
      <TabContent value="5">Contenido 5</TabContent>
      <TabContent value="6">Contenido 6</TabContent>
    </TabsProvider>
  )
}
