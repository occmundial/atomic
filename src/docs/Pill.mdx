# Pill

---

Group of buttons to toggle.

## Props

| Name             | Type                    | Default value    | Description                                                                                                                                                                 |
| ---------------- | ----------------------- | ---------------- | --------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| `label`          | **string**              | -                | Label text to show above the button's group.                                                                                                                                |
| `assistiveText`  | **string**              | -                | Assistive text to show below the button's group. This text turns into an error message when the component receives the error prop.                                          |
| `group`          | **Pill[]**              | -                | Array of items to render a bar of buttons. There just can be one selected item at a time. Each item must have an id (unique) and a label.                                   |
| `stack`          | **Pill[]**              | -                | Array of items to render a list of closable pills. Each item must have an id (unique) and a label.                                                                          |
| `choice`         | **Pill[]**              | -                | Array of items to render as a block of independent pills. Each of these pills is toggable. Each item must have an id (unique) and a label.                                  |
| `selected`       | **string, number**      | -                | Value to specify the selected item for group pills.                                                                                                                         |
| `onChange`       | **function**            | -                | Function to call when selecting an option in a group or a choice array.                                                                                                     |
| `onClick`        | **function**            | -                | Function to call when clicking on a stack item.                                                                                                                             |
| `onClose`        | **function**            | -                | Function to call when clicking on the close icon of a stack item.                                                                                                           |
| `error`          | **boolean**             | false            | Determines if the pill has an error state.                                                                                                                                  |
| `allowError`     | **boolean**             | false            | Determines if the error state is allowed without touching the pill.                                                                                                         |
| `disabled`       | **boolean**             | false            | Disable all group pills.                                                                                                                                                    |
| `lockHeight`     | **boolean**             | false            | By default the Pill container would be smaller without a label or an assistive text. lockHeight will maintain the same height as if it had both, label and assistive text.  |
| `idPrefix`       | **string**              | -                | Prefix for the id of every pill. If you don't specify a prefix they will not have an id.                                                                                    |
| `id`             | **string**              | -                | Id of the component                                                                                                                                                         |
| `className`      | **string**              | -                | Adds class(es) to each option container                                                                                                                                     |
| `style`          | **CSSProperties**       | -                | Adds style(s) to each option container                                                                                                                                      |
| `testId`         | **string**              | -                | The testId property adds the data attribute data-testid to the main element and should be used for testing only.                                                            |

## Examples

### Pill Group

<Demo title="Pill" background={colors.bgWhite}>
  <PillGroup />
</Demo>

```jsx
// How to use Pill Group
import React, { useState } from 'react'
import Pill from '@/components/Pill'

const pillsData = [
  {
    id: 1,
    value: '10',
    label: 'Element 1'
  },
  {
    id: 2,
    value: '20',
    label: 'Element 2'
  },
  {
    id: 3,
    value: '30',
    label: 'Element 3',
    disabled: true
  },
  {
    id: 4,
    value: '40',
    label: 'Element 4'
  }
]

const PillExample = () => {
  const [pills, _] = useState(pillsData)
  const [selected, setSelected] = useState(0)

  const onChange = (selected: number) => {
    setSelected(selected)
  }

  return (
    <Pill
      label="Label"
      group={pills}
      assistiveText="Assistive Text"
      onChange={onChange}
      selected={selected}
    />
  )
}
```

```jsx
// How to use Pill Choice
import React, { useState } from 'react'
import Pill from '@/components/Pill'

const PILLS_DUMMY = [
  {
    id: 1,
    value: 10,
    label: 'Label',
    leftIcon: 'person'
  },
  {
    id: 2,
    value: 20,
    label: 'Label',
    selected: true,
    leftIcon: 'person'
  },
  {
    id: 3,
    value: 30,
    label: 'Label',
    selected: true,
    disabled: true,
    leftIcon: 'person'
  },
  {
    id: 4,
    value: 40,
    label: 'Label',
    disabled: true,
    leftIcon: 'person'
  },
  {
    id: 5,
    value: 50,
    label: 'Label',
    leftIcon: 'person'
  }
]

const PillExample = () => {
  const [pills, setPills] = useState(PILLS_DUMMY)

  const onChange = (id: number) => {
    const updatedPills = pills.map(pill =>
      pill.id === id ? { ...pill, selected: !pill.selected } : pill
    )
    setPills(updatedPills)
  }

  return (
    <Pill
      label="Label"
      choice={pills}
      assistiveText="Assistive Text"
      onChange={onChange}
    />
  )
}
```
